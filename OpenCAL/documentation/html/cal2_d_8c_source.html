638  //Note that a substates' update is performed after each elementary process.
639 
640  int b, i, j, n;
641 
642  for (b=0; b<ca2D->num_of_elementary_processes; b++)
643  {
644  if (ca2D->A.cells) //Computationally active cells optimization.
645  for (n=0; n<ca2D->A.size; n++)
646  ca2D->elementary_processes[b](ca2D, ca2D->A.cells[n].i, ca2D->A.cells[n].j);
647  else //Standart cicle of the transition function
648  for (i=0; i<ca2D->rows; i++)
649  for (j=0; j<ca2D->columns; j++)
650  ca2D->elementary_processes[b](ca2D, i, j);
651 
652  //updating substates
653  calUpdate2D(ca2D);
654  }
